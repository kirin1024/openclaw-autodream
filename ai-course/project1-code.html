搭建一个可运行的意图识别Demo

1 环境准备

终端

2 后端：意图识别API

1

创建后端文件

新建文件 server.py

server.py 
# 导入FastAPI
from fastapi import FastAPI
from pydantic import BaseModel
from transformers import pipeline

# 创建应用
app = FastAPI(title="意图识别API")

# 加载模型（首次运行会下载）
print("正在加载模型...")
classifier = pipeline(
    "zero-shot-classification",
    model="facebook/bart-large-mnli"
)
print("模型加载完成！")

# 定义请求格式
class IntentRequest(BaseModel):
    text: str
    labels: list[str]

# 定义API接口
@app.post("/intent")
def predict_intent(request: IntentRequest):
    result = classifier(
        request.text,
        candidate_labels=request.labels
    )
    return {
        "intent": result["labels"][0],
        "confidence": result["scores"][0],
        "all_results": list(zip(result["labels"], result["scores"]))
    }

# 启动服务
if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
2

启动服务

# 终端运行
python server.py
✓ 服务已启动：http://localhost:8000
✓ API文档：http://localhost:8000/docs

3 前端：Vue界面

用Vue + Element Plus 做一个简单的界面。

App.vue 
<template>
  <div class="container">
    <h2>🤖 意图识别Demo</h2>
    
    <el-input
      v-model="userText"
      placeholder="请输入要识别的话术"
      type="textarea"
      :rows="3"
    ></el-input>
    
    <el-button 
      type="primary" 
      @click="recognizeIntent"
      :loading="loading"
    >
      开始识别
    </el-button>
    
    <div v-if="result" class="result-box">
      <h3>识别结果：</h3>
      <p>意图：{{ result.intent }}</p>
      <p>置信度：{{ (result.confidence * 100).toFixed(1) }}%</p>
    </div>
  </div>
</template>
<script setup>
import { ref } from 'vue'

const userText = ref('')
const loading = ref(false)
const result = ref(null)

const labels = ['查订单', '退货退款', '投诉', '咨询产品']

const recognizeIntent = async () => {
  loading.value = true
  try {
    const res = await fetch('http://localhost:8000/intent', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        text: userText.value,
        labels: labels
      })
    })
    result.value = await res.json()
  } catch (e) {
    console.error(e)
  }
  loading.value = false
}
</script>
